-- 1. Count total students
SELECT COUNT(*) AS total_students
FROM students;


-- 2. Find average age
SELECT AVG(age) AS average_age
FROM students;


-- 3. Find maximum age
SELECT MAX(age) AS max_age
FROM students;


-- 4. Find minimum age
SELECT MIN(age) AS min_age
FROM students;


-- 5. Count number of students in each branch
SELECT branch, COUNT(*) AS total_students
FROM students
GROUP BY branch;


-- 6. Find average age branch-wise
SELECT branch, AVG(age) AS avg_age
FROM students
GROUP BY branch;


-- 7. Display branches where total students > 1
SELECT branch, COUNT(*) AS total_students
FROM students
GROUP BY branch
HAVING COUNT(*) > 1;
